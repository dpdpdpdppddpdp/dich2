-- =====================================================
-- База данных для книжного магазина "ЧитайГород"
-- Демонстрационный экзамен 09.02.07
-- =====================================================

-- 1. Таблица ролей пользователей
CREATE TABLE IF NOT EXISTS roles (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL UNIQUE
);

-- 2. Таблица пользователей
CREATE TABLE IF NOT EXISTS users (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    full_name TEXT NOT NULL,
    login TEXT NOT NULL UNIQUE,
    password TEXT NOT NULL,
    role_id INTEGER NOT NULL,
    FOREIGN KEY (role_id) REFERENCES roles (id) ON DELETE RESTRICT
);

-- 3. Таблица товаров
CREATE TABLE IF NOT EXISTS products (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    article TEXT NOT NULL UNIQUE,
    name TEXT NOT NULL,
    unit TEXT DEFAULT 'шт.',
    price REAL NOT NULL CHECK (price >= 0),
    supplier TEXT,
    manufacturer TEXT,
    category TEXT,
    discount REAL DEFAULT 0 CHECK (discount BETWEEN 0 AND 100),
    stock_quantity INTEGER NOT NULL DEFAULT 0 CHECK (stock_quantity >= 0),
    description TEXT,
    photo_path TEXT
);

-- 4. Таблица пунктов выдачи
CREATE TABLE IF NOT EXISTS pickup_points (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    address TEXT NOT NULL UNIQUE
);

-- 5. Таблица заказов
CREATE TABLE IF NOT EXISTS orders (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    order_number INTEGER NOT NULL UNIQUE,
    order_date TEXT NOT NULL,
    delivery_date TEXT,
    pickup_point_id INTEGER NOT NULL,
    client_user_id INTEGER NOT NULL,
    pickup_code TEXT NOT NULL,
    status TEXT NOT NULL DEFAULT 'Новый',
    FOREIGN KEY (pickup_point_id) REFERENCES pickup_points (id),
    FOREIGN KEY (client_user_id) REFERENCES users (id)
);

-- 6. Таблица состава заказа (связь многие-ко-многим)
CREATE TABLE IF NOT EXISTS order_items (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    order_id INTEGER NOT NULL,
    product_article TEXT NOT NULL,
    quantity INTEGER NOT NULL DEFAULT 1 CHECK (quantity > 0),
    price_at_time REAL NOT NULL,
    discount_at_time REAL DEFAULT 0,
    FOREIGN KEY (order_id) REFERENCES orders (id) ON DELETE CASCADE,
    FOREIGN KEY (product_article) REFERENCES products (article) ON DELETE RESTRICT
);

-- 7. Таблица истории статусов заказов
CREATE TABLE IF NOT EXISTS order_status_history (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    order_id INTEGER NOT NULL,
    status TEXT NOT NULL,
    changed_at TEXT NOT NULL DEFAULT CURRENT_TIMESTAMP,
    changed_by INTEGER,
    FOREIGN KEY (order_id) REFERENCES orders (id) ON DELETE CASCADE,
    FOREIGN KEY (changed_by) REFERENCES users (id)
);

-- =====================================================
-- Индексы для ускорения запросов
-- =====================================================
CREATE INDEX IF NOT EXISTS idx_users_login ON users (login);
CREATE INDEX IF NOT EXISTS idx_users_role ON users (role_id);
CREATE INDEX IF NOT EXISTS idx_products_category ON products (category);
CREATE INDEX IF NOT EXISTS idx_products_manufacturer ON products (manufacturer);
CREATE INDEX IF NOT EXISTS idx_orders_user ON orders (client_user_id);
CREATE INDEX IF NOT EXISTS idx_orders_number ON orders (order_number);
CREATE INDEX IF NOT EXISTS idx_orders_status ON orders (status);
CREATE INDEX IF NOT EXISTS idx_order_items_order ON order_items (order_id);
CREATE INDEX IF NOT EXISTS idx_order_items_product ON order_items (product_article);

-- Проверочные запросы (для контроля)

-- Проверка количества записей
SELECT 'Пользователи' as table_name, COUNT(*) as count FROM users
UNION ALL
SELECT 'Товары', COUNT(*) FROM products
UNION ALL
SELECT 'Пункты выдачи', COUNT(*) FROM pickup_points
UNION ALL
SELECT 'Заказы', COUNT(*) FROM orders
UNION ALL
SELECT 'Состав заказов', COUNT(*) FROM order_items;

-- Вывод информации о пользователях с ролями
SELECT u.id, u.full_name, u.login, r.name as role_name
FROM users u
JOIN roles r ON u.role_id = r.id;

-- Вывод информации о заказах с названиями товаров
SELECT 
    o.order_number,
    o.order_date,
    o.status,
    GROUP_CONCAT(p.article || ' (' || p.name || ') x ' || oi.quantity, ', ') as products
FROM orders o
JOIN order_items oi ON o.id = oi.order_id
JOIN products p ON oi.product_article = p.article
GROUP BY o.id;
